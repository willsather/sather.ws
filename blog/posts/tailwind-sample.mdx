  say: "Hello" | "Goodbye";
  user: {
    firstName: string;
    lastName: string;
  };
}

/**
 * @param {"Hello" | "Goodbye} say what you want to say

export default function sayHello(say: "Hello" | "Goodbye"): SayHelloType {

    say: say,

  };
}
